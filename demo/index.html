Video Pointer

JavaScript
$("#vp1").videopointer({
    points:[
        {
            x:47,
            y:63,
            color:"#9BC9BD",
            time:12
        },
        {
            x:50,
            y:35,
            color:"#A2B5C4",
            time:1
        }
    ]
});
